# 配列操作

作成日 2021/01/24

```javascript
const arr = ['a', 'b', 'c'];
let res = '';

// 配列の最初の要素を削除し、削除した要素を返す
res = arr.shift();
//=> arr = ['b', 'c']
//=> res = 'a'

// 配列の最後の要素を削除し、削除した要素を返す
res = arr.pop();
//=> arr = ['b']
//=> res = 'c'

// 配列の先頭に1つ以上の要素を追加し、新しい配列の長さを返す
res = arr.unshift('x', 'y', 'z');
//=> arr = ['x', 'y', 'z', 'b']
//=> res = 4

// 配列の最後に1つ以上の要素を追加し、新しい配列の長さを返す
res = arr.push('d', 'e', 'f');
//=> arr = ['x', 'y', 'z', 'b', 'd', 'e', 'f']
//=> res = 7

// 0から数えて、a番目からb個の要素を削除し、代わりに要素を埋め込み、
// 削除した要素を返す
res = arr.splice(3, 3, 'g', 'h', 'i');
//=> arr = ['x', 'y', 'z', 'g', 'h', 'i', 'f']
//=> res = ['b', 'd', 'e']
```
