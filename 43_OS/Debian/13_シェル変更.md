# シェルを変更する

作成日 2025/12/24

```bash
# 現在のシェルを表示する
echo $SHELL
# /bin/sh

# 利用可能なシェルの一覧を表示する
cat /etc/shells
# # /etc/shells: valid login shells
# /bin/sh
# /usr/bin/sh
# /bin/bash
# /usr/bin/bash
# /bin/rbash
# /usr/bin/rbash
# /usr/bin/dash

# 一時的にシェルを変更する
bash

# 現在のユーザーのシェルを変更する
chsh -s /bin/bash
# Password:
```
