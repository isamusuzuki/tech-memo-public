# パス文字列を操作する

作成日 2025/11/20

```python
import os

filepath = './dir/subdir/filename.ext'

# ファイル名を取得する
basename = os.path.basename(filepath)
# => filename.ext

# 拡張子なしのファイル名を取得する
basename_no_ext = os.path.splittext(os.path.basename(filepath))[0]
# => filename

# フォルダ名を取得する
dirname = os.path.dirname(filepath)
# => ./dir/subdir

# ファイル名とフォルダ名を分ける
base_dir_pair = os.path.split(filepath)
# => ('./dir/subdir', 'filename.ext')

# 拡張子を取得する
root_ext_pair = os.path.splittext(filepath)
# => ('./dir/subdir/filename', 'ext')

# ファイル名とフォルダ名を結合してパス文字列を作成する
path = os.path.join('dir', 'subdir', 'filename.ext')
# => ./dir/subdir/filename.ext
```
