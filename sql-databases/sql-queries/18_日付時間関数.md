# 日付時間関数

作成日 2020/11/10、更新日 2021/06/18

## 01. UTC で入っている TIMESTAMP を 日本時間に直して表示する

```sql
SELECT
    sku_code,
    overwrite_times,
    DATE_ADD(updated_at, INTERVAL 9 HOUR) updated_at_jpn,
    DATE_ADD(created_at, INTERVAL 9 HOUR) created_at_jpn
FROM
    sku_images
ORDER BY
    updated_at DESC
LIMIT 10;
```

## 02. 締め日を SQL クエリーだけで求める

```sql
-- 現在の日付
SELECT
    CURRENT_DATE;
-- 2020-04-15

-- 今月16日の深夜0時
SELECT
    DATE_FORMAT(CURRENT_DATE, '%Y-%m-16 00:00:00');
-- 2020-04-16 00:00:00

-- 今月16日の深夜0時のUTC
SELECT
    CONVERT_TZ(
        DATE_FORMAT(CURRENT_DATE, '%Y-%m-16 00:00:00'),
        '+09:00',
        '+00:00'
    );
-- 2020-04-15 15:00:00

-- 今月16日の深夜0時のUTCのタイムスタンプ
SELECT
    UNIX_TIMESTAMP(
        CONVERT_TZ(
            DATE_FORMAT(CURRENT_DATE, '%Y-%m-16 00:00:00'),
            '+09:00',
            '+00:00'
        )
    );
-- 1586962800

-- 今月16日の深夜0時のUTCのタイムスタンプの千倍
SELECT
    CAST(
        UNIX_TIMESTAMP(
            CONVERT_TZ(
                DATE_FORMAT(CURRENT_DATE, '%Y-%m-16 00:00:00'),
                '+09:00',
                '+00:00'
            )
        ) AS SIGNED
    ) * 1000;
-- 1586962800000

-- テスト
SELECT
    COUNT(*)
FROM
    Posts
WHERE
    ChannelId = 'my7erexuapgyz8z9397pdd5w9c'
    AND CreateAt < CAST(
        UNIX_TIMESTAMP(
            CONVERT_TZ(
                DATE_FORMAT(CURRENT_DATE, '%Y-%m-16 00:00:00'),
                '+09:00',
                '+00:00'
            )
        ) AS SIGNED
    ) * 1000;
-- 成功
```

## 03. 開始日を SQL クエリーだけで求める

```sql
-- 現在の日付
SELECT
    CURRENT_DATE;
-- 2020-04-15

-- 現在から1ヶ月前
SELECT
    DATE_SUB(CURRENT_DATE, INTERVAL 1 MONTH);
-- 2020-03-15

-- 先月16日の深夜0時
SELECT
    DATE_FORMAT(
        DATE_SUB(CURRENT_DATE, INTERVAL 1 MONTH),
        '%Y-%m-16 00:00:00'
    );
-- 2020-03-16 00:00:00

-- 先月16日の深夜0時のUTC
SELECT
    CONVERT_TZ(
        DATE_FORMAT(
            DATE_SUB(CURRENT_DATE, INTERVAL 1 MONTH),
            '%Y-%m-16 00:00:00'
        ),
        '+09:00',
        '+00:00'
    );
-- 2020-03-15 15:00:00

-- 先月16日の深夜0時のUTCのタイムスタンプ
SELECT
    UNIX_TIMESTAMP(
        CONVERT_TZ(
            DATE_FORMAT(
                DATE_SUB(CURRENT_DATE, INTERVAL 1 MONTH),
                '%Y-%m-16 00:00:00'
            ),
            '+09:00',
            '+00:00'
        )
    );
-- 1584284400

-- 先月16日の深夜0時のUTCのタイムスタンプの千倍
SELECT
    CAST(
        UNIX_TIMESTAMP(
            CONVERT_TZ(
                DATE_FORMAT(
                    DATE_SUB(CURRENT_DATE, INTERVAL 1 MONTH),
                    '%Y-%m-16 00:00:00'
                ),
                '+09:00',
                '+00:00'
            )
        ) AS SIGNED
    ) * 1000;
-- 1584284400000

-- テスト
SELECT
    COUNT(*)
FROM
    Posts
WHERE
    ChannelId = 'my7erexuapgyz8z9397pdd5w9c'
    AND CreateAt >= CAST(
        UNIX_TIMESTAMP(
            CONVERT_TZ(
                DATE_FORMAT(
                    DATE_SUB(CURRENT_DATE, INTERVAL 1 MONTH),
                    '%Y-%m-16 00:00:00'
                ),
                '+09:00',
                '+00:00'
            )
        ) AS SIGNED
    ) * 1000
    AND CreateAt < CAST(
        UNIX_TIMESTAMP(
            CONVERT_TZ(
                DATE_FORMAT(CURRENT_DATE, '%Y-%m-16 00:00:00'),
                '+09:00',
                '+00:00'
            )
        ) AS SIGNED
    ) * 1000;
-- 成功
```
